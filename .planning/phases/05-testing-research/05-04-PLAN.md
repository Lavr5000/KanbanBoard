import { describe, it, expect, vi } from 'vitest'
import { render, screen, fireEvent, waitFor } from '@testing-library/react'
import userEvent from '@testing-library/user-event'
import { AddTaskModal } from '@/features/task-operations/ui/AddTaskModal'

describe('AddTaskModal Component', () => {
  const mockProps = {
    isOpen: true,
    onClose: vi.fn(),
    columnId: 'col-1',
    onTaskAdded: vi.fn(),
  }

  it('renders when isOpen is true', () => {
    render(<AddTaskModal {...mockProps} />)
    expect(screen.getByText(/add task|new task|создать задачу/i)).toBeInTheDocument()
  })

  it('does not render when isOpen is false', () => {
    render(<AddTaskModal {...mockProps} isOpen={false} />)
    expect(screen.queryByText(/add task|new task/i)).not.toBeInTheDocument()
  })

  it('calls onClose when cancel button is clicked', async () => {
    const user = userEvent.setup()
    render(<AddTaskModal {...mockProps} />)

    const cancelButton = screen.getByRole('button', { name: /cancel|отмена/i })
    await user.click(cancelButton)

    expect(mockProps.onClose).toHaveBeenCalledTimes(1)
  })

  it('validates empty task content', async () => {
    const user = userEvent.setup()
    render(<AddTaskModal {...mockProps} />)

    // Try to save without content
    const saveButton = screen.getByRole('button', { name: /save|сохранить/i })
    await user.click(saveButton)

    // Should show validation error
    expect(screen.getByText(/required|обязательно/i)).toBeInTheDocument()
  })

  it('calls onTaskAdded with correct data', async () => {
    const user = userEvent.setup()
    render(<AddTaskModal {...mockProps} />)

    const input = screen.getByRole('textbox')
    await user.type(input, 'New test task')

    const saveButton = screen.getByRole('button', { name: /save|сохранить/i })
    await user.click(saveButton)

    await waitFor(() => {
      expect(mockProps.onTaskAdded).toHaveBeenCalledWith({
        content: 'New test task',
        columnId: 'col-1',
        priority: expect.any(String),
      })
    })
  })

  it('allows priority selection', async () => {
    const user = userEvent.setup()
    render(<AddTaskModal {...mockProps} />)

    const highPriorityButton = screen.getByRole('button', { name: /high|срочно/i })
    await user.click(highPriorityButton)

    // Verify selection state
    expect(highPriorityButton).toHaveClass(/selected|active|bg-red/i)
  })

  it('works with React 19 concurrent features', async () => {
    const user = userEvent.setup()
    render(<AddTaskModal {...mockProps} />)

    // Rapid state changes (tests React 19 concurrent rendering)
    const input = screen.getByRole('textbox')

    await user.type(input, 'A')
    await user.type(input, 'B')
    await user.type(input, 'C')

    expect(input).toHaveValue('ABC')
  })
})

import { describe, it, expect, vi } from 'vitest'
import { render, screen, waitFor } from '@testing-library/react'
import userEvent from '@testing-library/user-event'
import { EditTaskModal } from '@/features/task-operations/ui/EditTaskModal'

describe('EditTaskModal Component', () => {
  const mockTask = {
    id: '1',
    content: 'Original task content',
    columnId: 'col-1',
    priority: 'medium' as const,
    createdAt: new Date().toISOString(),
    tags: ['tag1', 'tag2'],
  }

  const mockProps = {
    isOpen: true,
    onClose: vi.fn(),
    task: mockTask,
    onTaskUpdated: vi.fn(),
  }

  it('pre-fills with task data', () => {
    render(<EditTaskModal {...mockProps} />)

    const input = screen.getByRole('textbox')
    expect(input).toHaveValue('Original task content')
  })

  it('pre-selects correct priority', () => {
    render(<EditTaskModal {...mockProps} />)

    const mediumPriority = screen.getByRole('button', { name: /medium|обычно/i })
    expect(mediumPriority).toHaveClass(/selected|active|bg-amber/i)
  })

  it('updates task content', async () => {
    const user = userEvent.setup()
    render(<EditTaskModal {...mockProps} />)

    const input = screen.getByRole('textbox')
    await user.clear(input)
    await user.type(input, 'Updated task content')

    const saveButton = screen.getByRole('button', { name: /save|сохранить/i })
    await user.click(saveButton)

    await waitFor(() => {
      expect(mockProps.onTaskUpdated).toHaveBeenCalledWith({
        ...mockTask,
        content: 'Updated task content',
      })
    })
  })

  it('displays existing tags', () => {
    render(<EditTaskModal {...mockProps} />)

    expect(screen.getByText('tag1')).toBeInTheDocument()
    expect(screen.getByText('tag2')).toBeInTheDocument()
  })

  it('allows adding new tags', async () => {
    const user = userEvent.setup()
    render(<EditTaskModal {...mockProps} />)

    const tagInput = screen.getByPlaceholderText(/add tag|добавить тег/i)
    await user.type(tagInput, 'newtag{Enter}')

    expect(screen.getByText('newtag')).toBeInTheDocument()
  })

  it('removes tags when clicked', async () => {
    const user = userEvent.setup()
    render(<EditTaskModal {...mockProps} />)

    const tag1 = screen.getByText('tag1')
    await user.click(tag1)

    expect(screen.queryByText('tag1')).not.toBeInTheDocument()
  })
})

import { describe, it, expect, vi } from 'vitest'
import { render, screen } from '@testing-library/react'
import { ExportModal } from '@/features/export-data/ui/ExportModal'

describe('ExportModal Component', () => {
  const mockProps = {
    isOpen: true,
    onClose: vi.fn(),
    columns: [],
    tasks: [],
  }

  it('renders export options', () => {
    render(<ExportModal {...mockProps} />)

    expect(screen.getByText(/export/i)).toBeInTheDocument()
    expect(screen.getByText(/pdf/i)).toBeInTheDocument()
    expect(screen.getByText(/excel|xlsx/i)).toBeInTheDocument()
  })

  it('has correct z-index to prevent Bug #1', () => {
    render(<ExportModal {...mockProps} />)

    const modal = screen.getByText(/export/i).closest('#modal-overlay')
    expect(modal?.style.zIndex).toBe('99999')
  })

  it('disables export when no tasks available', () => {
    render(<ExportModal {...mockProps} columns={[]} tasks={[]} />)

    const exportButtons = screen.getAllByRole('button')
    exportButtons.forEach(button => {
      if (button.textContent?.match(/export/i)) {
        expect(button).toBeDisabled()
      }
    })
  })

  it('enables export when tasks are available', () => {
    const propsWithTasks = {
      ...mockProps,
      columns: [{ id: '1', title: 'Col 1', boardId: 'board-1' }],
      tasks: [{ id: '1', content: 'Task', columnId: '1', priority: 'high', createdAt: '' }],
    }

    render(<ExportModal {...propsWithTasks} />)

    const exportButton = screen.getByRole('button', { name: /export pdf/i })
    expect(exportButton).not.toBeDisabled()
  })

  it('calls onClose when backdrop clicked', () => {
    render(<ExportModal {...mockProps} />)

    const backdrop = screen.getByText(/export/i).closest('#modal-overlay')
    if (backdrop) {
      backdrop.click()
      expect(mockProps.onClose).toHaveBeenCalled()
    }
  })
})

import { describe, it, expect, vi } from 'vitest'
import { render, screen } from '@testing-library/react'
import { BoardBackground } from '@/widgets/board/ui/BoardBackground'

describe('BoardBackground Component', () => {
  it('renders without crashing', () => {
    render(<BoardBackground />)
    expect(screen.getByTestId('board-background')).toBeInTheDocument()
  })

  it('renders animated gradient', () => {
    render(<BoardBackground />)
    const gradient = screen.getByTestId('animated-gradient')
    expect(gradient).toHaveClass(/animated-gradient/i)
  })

  it('renders noise texture overlay', () => {
    render(<BoardBackground />)
    const noise = screen.getByTestId('noise-texture')
    expect(noise).toHaveStyle({ opacity: expect.stringContaining('0.03') })
  })

  it('renders 4 glow orbs', () => {
    render(<BoardBackground />)
    const orbs = screen.getAllByTestId(/glow-orb/i)
    expect(orbs).toHaveLength(4)
  })

  it('cleans up event listeners on unmount', () => {
    const { unmount } = render(<BoardBackground />)

    // Spy on removeEventListener
    const removeSpy = vi.spyOn(window, 'removeEventListener')

    unmount()

    // Verify cleanup happened
    expect(removeSpy).toHaveBeenCalledWith(
      'mousemove',
      expect.any(Function)
    )
  })

  it('does not cause memory leaks', () => {
    const { rerender } = render(<BoardBackground />)

    // Multiple re-renders should not cause issues
    for (let i = 0; i < 10; i++) {
      rerender(<BoardBackground />)
    }

    // Component should still render correctly
    expect(screen.getByTestId('board-background')).toBeInTheDocument()
  })
})

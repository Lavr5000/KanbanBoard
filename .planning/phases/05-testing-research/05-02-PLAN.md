import { test, expect } from 'playwright/test'

test.describe('Card Lifecycle E2E', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/')
    await page.waitForSelector('[data-testid="board-container"]')
  })

  test('should create a new task', async ({ page }) => {
    // Click add task button
    await page.click('[data-testid="add-task-button"]')

    // Fill in task content
    await page.fill('[data-testid="task-content-input"]', 'New E2E test task')

    // Select priority
    await page.click('[data-testid="priority-high"]')

    // Save task
    await page.click('[data-testid="save-task-button"]')

    // Verify task appears in column
    await expect(page.locator('text=New E2E test task')).toBeVisible()
  })

  test('should edit an existing task', async ({ page }) => {
    // Click on first task to edit
    await page.click('[data-testid="task-card"]:first-child')

    // Edit content
    const contentInput = page.locator('[data-testid="task-content-input"]')
    await contentInput.clear()
    await contentInput.fill('Updated task content')

    // Save
    await page.click('[data-testid="save-task-button"]')

    // Verify update
    await expect(page.locator('text=Updated task content')).toBeVisible()
  })

  test('should delete a task', async ({ page }) => {
    // Get initial task count
    const initialCount = await page.locator('[data-testid="task-card"]').count()

    // Delete first task
    await page.click('[data-testid="task-card"]:first-child [data-testid="delete-task-button"]')

    // Confirm deletion
    await page.click('[data-testid="confirm-delete-button"]')

    // Verify task count decreased
    const newCount = await page.locator('[data-testid="task-card"]').count()
    expect(newCount).toBe(initialCount - 1)
  })

  test('should move task between columns via drag and drop', async ({ page }) => {
    // Get initial column counts
    const sourceColumn = page.locator('[data-testid="column"]:first-child')
    const targetColumn = page.locator('[data-testid="column"]:nth-child(2)')

    const sourceInitialCount = await sourceColumn.locator('[data-testid="task-card"]').count()
    const targetInitialCount = await targetColumn.locator('[data-testid="task-card"]').count()

    // Drag first task from first column to second column
    const firstTask = sourceColumn.locator('[data-testid="task-card"]:first-child')

    await firstTask.dragTo(targetColumn)

    // Verify counts changed
    const sourceNewCount = await sourceColumn.locator('[data-testid="task-card"]').count()
    const targetNewCount = await targetColumn.locator('[data-testid="task-card"]').count()

    expect(sourceNewCount).toBe(sourceInitialCount - 1)
    expect(targetNewCount).toBe(targetInitialCount + 1)
  })
})

import { test, expect } from '@playwright/test'

test.describe('Modal Visual Regression (Z-Index Bug Prevention)', () => {
  test('Export modal appears on top of all content', async ({ page }) => {
    await page.goto('/')

    // Wait for board to load
    await page.waitForSelector('[data-testid="board-container"]')

    // Open export modal
    await page.click('[data-testid="export-button"]')

    // Wait for modal to appear
    await page.waitForSelector('#modal-overlay')

    // Take screenshot for visual regression
    await expect(page).toHaveScreenshot('export-modal-visible.png', {
      maxDiffPixels: 100,
    })

    // Verify modal is visible (not hidden behind columns)
    const modal = page.locator('#modal-overlay')
    await expect(modal).toBeVisible()

    // Verify modal z-index is higher than columns
    const modalZIndex = await modal.evaluate(el =>
      window.getComputedStyle(el).zIndex
    )
    const columnZIndex = await page.locator('[data-testid="column"]').first().evaluate(el =>
      window.getComputedStyle(el).zIndex
    )

    expect(parseInt(modalZIndex)).toBeGreaterThan(parseInt(columnZIndex || '0'))
  })

  test('Add task modal appears on top', async ({ page }) => {
    await page.goto('/')

    // Open add task modal
    await page.click('[data-testid="add-task-button"]')

    // Screenshot
    await expect(page).toHaveScreenshot('add-task-modal-visible.png', {
      maxDiffPixels: 100,
    })

    // Verify modal overlay covers entire screen
    const overlay = page.locator('#modal-overlay')
    const box = await overlay.boundingBox()

    expect(box?.width).toBeGreaterThanOrEqual(window.innerWidth)
    expect(box?.height).toBeGreaterThanOrEqual(window.innerHeight)
  })

  test('Modal backdrop closes modal on click', async ({ page }) => {
    await page.goto('/')

    // Open modal
    await page.click('[data-testid="add-task-button"]')
    await page.waitForSelector('#modal-overlay')

    // Click backdrop (outside modal content)
    const overlay = page.locator('#modal-overlay')
    await overlay.click({ position: { x: 10, y: 10 } })

    // Modal should close
    await expect(overlay).not.toBeVisible()
  })

  test('Multiple modals stack correctly', async ({ page }) => {
    await page.goto('/')

    // Open first modal
    await page.click('[data-testid="add-task-button"]')
    await page.waitForSelector('#modal-overlay')

    // Open second modal (e.g., from within first modal)
    // This tests modal-on-modal stacking

    // Screenshot to verify visual stacking
    await expect(page).toHaveScreenshot('modal-stacking.png', {
      maxDiffPixels: 150,
    })
  })
})

import { test, expect } from '@playwright/test'

test.describe('React 19 Compatibility', () => {
  test('no React 19 internal errors in console', async ({ page }) => {
    const consoleErrors: string[] = []

    page.on('console', msg => {
      if (msg.type() === 'error') {
        consoleErrors.push(msg.text())
      }
    })

    await page.goto('/')

    // Interact with various components
    await page.click('[data-testid="add-task-button"]')
    await page.fill('[data-testid="task-content-input"]', 'Test task')
    await page.click('[data-testid="cancel-task-button"]')

    // Check for React 19 specific errors
    const reactErrors = consoleErrors.filter(err =>
      err.includes('Expected static flag') ||
      err.includes('React 19') ||
      err.includes('concurrent')
    )

    expect(reactErrors).toHaveLength(0)
  })

  test('concurrent features work correctly', async ({ page }) => {
    await page.goto('/')

    // Test concurrent state updates
    await page.click('[data-testid="add-task-button"]')

    // Rapid inputs (should be handled by React 19 concurrent rendering)
    await page.fill('[data-testid="task-content-input"]', 'A')
    await page.fill('[data-testid="task-content-input"]', 'AB')
    await page.fill('[data-testid="task-content-input"]', 'ABC')

    // Should not cause any issues
    const input = page.locator('[data-testid="task-content-input"]')
    await expect(input).toHaveValue('ABC')
  })

  test('actions and useOptimistic work', async ({ page }) => {
    await page.goto('/')

    // Test optimistic updates (if implemented)
    // This ensures React 19 new features work
    const taskCountBefore = await page.locator('[data-testid="task-card"]').count()

    // Quick action that might use optimistic updates
    await page.click('[data-testid="add-task-button"]')
    await page.fill('[data-testid="task-content-input"]', 'Optimistic test')
    await page.click('[data-testid="save-task-button"]')

    // Task should appear (optimistically or for real)
    await page.waitForTimeout(100)
    const taskCountAfter = await page.locator('[data-testid="task-card"]').count()

    expect(taskCountAfter).toBeGreaterThan(taskCountBefore)
  })
})
